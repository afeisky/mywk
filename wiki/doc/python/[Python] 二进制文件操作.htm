Python: 二进制文件操作

 

#!/usr/bin/python
#coding:utf-8

fin=open("userdate.bin",'rb')
byte=fin.read()
fin.close()

f = open("userdate.img", 'rb+')    #  二进制读写模式
f.seek(0, 2)    # 指针移到文件到尾,第二个参数2是文件尾
file_end = f.tell()     # 文件结尾位置号
f.seek(0,0)     # 指针移到文件到头
while f.tell() < file_end:
    f.seek(1024-1, 1)  # 指针在当前位置再移动1024个字节,第二个参数1是当前位置
    f.write(chr(0xff))
f.close()
